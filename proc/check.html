    subroutine check(status)

        use netcdf
        implicit none
        integer, intent ( in) :: status

        if(status /= nf90_noerr) then
            write(*,*) 'Stopping due to netcdf error: '//trim(nf90_strerror(status))
            error stop
        end if

    end subroutine check

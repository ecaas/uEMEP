    subroutine phih_func(z,L,phih,phih_i)

        implicit none

        real, intent(in) :: z,L
        real, intent(out) :: phih,phih_i
        real a,b,q
        parameter (a=16.,b=5.,q=-0.5)
        real eps

        eps=z/L

        if (eps.ge.0) then
            phih=1.+b*eps
            phih_i=-b*eps
        else
            phih=exp(q*log((1.-a*eps)))
            phih_i=2.*log((1.+1./phih)/2.)
        endif

    end subroutine phih_func
